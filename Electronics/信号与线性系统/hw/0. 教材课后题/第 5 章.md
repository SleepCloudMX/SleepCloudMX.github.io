   y_3(n) &= \dfrac{3 - 3^{-n}}{2} u(n) - \dfrac{3 - 3^{5-n}}{2} u(n-5)
   \\
   &= \dfrac{3 - 3^{-n}}{2} \bqty{u(n) - u(n-5)} + \dfrac{121}{3^n} u(n-5).
