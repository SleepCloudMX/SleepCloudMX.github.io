\begin{align}
p_{ij}^{(m + n)}
&=
P\Bqty{
	X_{m + n} = j \mid X_0 = i
} \\
&= \sum_{k \in S} P\Bqty{
	X_m = k, X_{m + n} = j \mid X_0 = i
}
\\
&= \sum_{k \in S} P\Bqty{
	X_m = k \mid X_0 = i
} P\Bqty{
	X_{m + n} = j \mid X_0 = i, X_m = k
} \\
&= \sum_{k \in S} p_{ik}^{(m)} p_{kj}^{(n)}.
\end{align}
